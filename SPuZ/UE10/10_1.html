# max delay of response
max.response = 40

# init 
a1 = matrix(c(0.5,0.1,0,0.5),  2, 2)
a2 = matrix(c(0,0.1,-0.9,0.2), 2, 2)
b = array(NA, c(max.response,2,2))

# first steps
b0  = matrix(c(1,0,0,1), 2, 2)
b[1,,] = a1%*%b0
b[2,,] = a1%*%b[1,,] + a2%*%b0

# rest
for (i in 3:max.response) {
  b[i,,] = a1 %*% b[i-1,,] + a2 %*% b[i-2,,]
}

for (i in 0:10) {
  print(b[i,,])
}
